    <h1>we will show latest 5 post with 2 custom meta info if available: prowp_product_type and prowp_price</h1>
    <?php
    $args = array(
        'posts_per_page' => 5,
        'meta_query' => array(
            'relation' => 'OR',
            array(
                'key' => '_prowp_type',
                'compare' => 'EXISTS'
            ),
            array(
                'key' => '_prowp_price',
                'compare' => 'EXISTS'
            )
        ),
        'orderby' => array(
            '_prowp_type' => 'DESC',
            'date' => 'DESC'
        )
    );
    $query = new WP_Query($args);
    if ($query->have_posts()) {
        while ($query->have_posts()) {
            $query->the_post();
            echo '<h2>' . get_the_title() . '</h2>';
            echo '<p>Product Type: ' . get_post_meta(get_the_ID(), '_prowp_type', true) . '</p>';
            echo '<p>Price: ' . get_post_meta(get_the_ID(), '_prowp_price', true) . '</p>';
        }
    } else {
        echo '<p>No posts found.</p>';
    }
    wp_reset_postdata();
